\section{Part 1 --- Project Overview}\label{part-1-project-overview}

\begin{frame}{Background and Motivation}
\phantomsection\label{background-and-motivation}
\begin{itemize}
\tightlist
\item
  Indonesia faces a growing NCD burden driven by population aging, high
  smoking prevalence, rising obesity, and limited screening access
\item
  CVD and cancer together account for over 50\% of premature mortality
  in Indonesia
\item
  Policy decisions require quantitative evidence on intervention
  scale-up trajectories and their expected health and economic returns
\end{itemize}

\begin{keyinsight}[Why Markov?]
Unlike system dynamics approaches (e.g., NCDSim), discrete-time Markov models allow transparent, state-by-state tracking of disease progression---making calibration to GBD and validation against observed rates straightforward.
\end{keyinsight}
\end{frame}

\begin{frame}{Aims and Scope}
\phantomsection\label{aims-and-scope}
\textbf{Primary aim:} Project health and economic outcomes of NCD policy
packages in Indonesia from 2025 to 2050.

\medskip

\textbf{Two independent disease modules:}

\begin{enumerate}
\tightlist
\item
  \textbf{CVD} --- IHD, ischemic stroke, hemorrhagic stroke,
  hypertensive heart disease
\item
  \textbf{Cancer} --- 16 types from GBD 2021, in three analytic tiers
\end{enumerate}

\medskip

\textbf{Policy levers modeled:}

\begin{itemize}
\tightlist
\item
  Hypertension control scale-up (HEARTS)
\item
  Sodium reduction
\item
  Trans-fat elimination
\item
  Statin coverage expansion
\item
  Cancer screening (cervical, colorectal, breast)
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Reproducible Research Workflow}
\phantomsection\label{reproducible-research-workflow}
\begin{center}
\textbf{Inputs} $\rightarrow$ \textbf{Scenario Definitions} $\rightarrow$ \textbf{Model Outputs} $\rightarrow$ \textbf{Documentation} $\rightarrow$ \textbf{Reports}
\end{center}

\bigskip

\begin{longtable}[]{@{}ll@{}}
\toprule\noalign{}
Stage & Source \\
\midrule\noalign{}
\endhead
Raw data & GBD 2023, UNWPP 2024, IHME, NCD-RisC, ETIHAD \\
Scenarios & \texttt{04\_define\_interventions.R} /
\texttt{settings.yml} \\

Post-processing & DALYs, VSL/VSLY calculations \\

\bottomrule\noalign{}
\end{longtable}

\begin{methodsnote}
All scenario definitions are code-driven and version-controlled---no manual spreadsheet steps in the analytic pipeline.
\end{methodsnote}
\end{frame}

\end{keyinsight}

\begin{itemize}
\tightlist
\item

  \texttt{data/processed}
\item

\item

  Intermediate \texttt{.rds} checkpoints at each pipeline stage

\begin{keyinsight}

\end{keyinsight}

\begin{frame}{Core Model Logic: CVD}
\phantomsection\label{core-model-logic-cvd}
\textbf{State space} (per age / sex / location / cause / year):

\[\text{Susceptible} \rightarrow \text{Incident} \rightarrow \text{Prevalent} \rightarrow \text{Dead}\]

\medskip

\begin{itemize}
\tightlist
\item
  Transition probabilities calibrated to GBD 2023 incidence, prevalence,
  and mortality
\item
  Background mortality layered from UNWPP 2024 life tables
\item
  Secular trends in case fatality forecast via ARIMA (\texttt{forecast}
  package)
\end{itemize}

\begin{methodsnote}
Constraint enforced: $IR + BG_{mx} \leq 1$ for all age--sex--cause--year cells. Violations are flagged in \texttt{032\_adjustments.R} and corrected before simulation.
\end{methodsnote}
\end{frame}

\begin{frame}{Core Model Logic: Cancer}
\phantomsection\label{core-model-logic-cancer}
\textbf{7-state Markov model} (per age / sex / location / cause / year):

\[\text{Well} \rightarrow \text{Precancer} \rightarrow \text{Local} \rightarrow \text{Regional} \rightarrow \text{Distant} \rightarrow \text{Cancer Death}\]

with absorbing background death state.

\medskip

\textbf{Three analytic tiers:}

\begin{enumerate}
\tightlist
\item
  \textbf{Tier 1} (11 cancers) --- Incidence/mortality only (no staging)
\item
  \textbf{Tier 2} (3 cancers) --- Staged: local, regional, distant
\item
  \textbf{Tier 3} (2 cancers) --- Precancerous lesions modeled (cervix,
  CRC)
\end{enumerate}

\begin{methodsnote}
Tier 3 cancers include screening interventions that shift stage distribution at diagnosis, directly affecting transition probabilities.
\end{methodsnote}
\end{frame}

\begin{frame}{Intervention Logic}
\phantomsection\label{intervention-logic}
\textbf{CVD interventions} are applied multiplicatively to incidence
rates:

\[IR_{\text{treated}} = IR_{\text{baseline}} \times (1 - \text{coverage} \times RR_{\text{effect}})\]

\medskip

\textbf{Multi-intervention stacking order:}

\begin{enumerate}
\tightlist
\item
  Sodium \(\rightarrow\) BP distribution shift via
  \texttt{get.bp.prob()} coefficients
\item
  Statins \(\rightarrow\) direct IR reduction (RR from meta-analyses)
\item
  TFA elimination \(\rightarrow\) IR reduction via \(RR = 1.28\) per 2\%
  energy from TFA
\item
  HTN control \(\rightarrow\) ETIHAD-based RR per 10 mmHg BP reduction
\end{enumerate}

\medskip

\textbf{Cancer interventions:} Screening coverage expansion shifts stage
distribution at diagnosis, reducing late-stage presentation and
improving survival.
\end{frame}

\begin{frame}{Economic Evaluation Framework}
\phantomsection\label{economic-evaluation-framework}
\textbf{Value of Statistical Life (VSL)} approach
(\citeproc{ref-Robinson2019VSL}{Robinson et al., 2019}):

\[VSL_{\text{country}} = VSL_{\text{base}} \times \left(\frac{GNI_{\text{country}}}{GNI_{\text{base}}}\right)^{\varepsilon}\]

\medskip

\begin{itemize}
\tightlist
\item
  Three income elasticity scenarios: \(\varepsilon = 1.0, 1.2, 1.5\)
  (primary: 1.2)
\item
  VSLY derived: \(VSLY = VSL / \bar{e}\) (average adult life expectancy)
\item
  GDP growth projections from SSP2 scenario
\end{itemize}

\begin{takeaway}
Economic valuation converts deaths averted and DALYs prevented into monetary terms, enabling comparison across intervention packages and informing resource allocation decisions.
\end{takeaway}
\end{frame}

\begin{frame}{How Modules Integrate}
\phantomsection\label{how-modules-integrate}
\textbf{Current state:} CVD and cancer pipelines run independently,
producing separate outputs (deaths averted, DALYs, economic value).

\medskip

\textbf{Integration points:}

\begin{itemize}
\tightlist
\item
  Shared demographic inputs (UNWPP 2024 population, life tables)
\item
  Shared economic framework (VSL/VSLY methodology)
\item
  Common reporting templates (Beamer, R Markdown)
\end{itemize}

\medskip

\textbf{Not yet integrated:}

\begin{itemize}
\tightlist
\item
  No joint population projection feeding both models
\item
  No comorbidity or competing-risk adjustment across CVD and cancer
\item
  No unified scenario configuration file
\end{itemize}

\begin{cautionbox}[Integration Pending]
A formal integration layer---shared population module, unified scenario config, and joint output aggregation---is planned for v2.0.
\end{cautionbox}
\end{frame}

\begin{frame}{CVD: Data Loading and Calibration}
\phantomsection\label{cvd-data-loading-and-calibration}
\textbf{Data sources ingested:}

\begin{itemize}
\tightlist
\item
  GBD 2023: cause-specific mortality, incidence, prevalence (4 CVD
  causes)
\item
  UNWPP 2024: population by age/sex, life expectancy, fertility
\item
  NCD-RisC: historical blood pressure distributions (1990--2019)
\item
  IHME DIET: sodium intake estimates
\end{itemize}

\medskip

\textbf{Calibration pipeline:}

\begin{enumerate}
\tightlist
\item
  Extract single-year age rates (20--95) from 5-year GBD groups via
  interpolation
\item
  Compute transition probabilities from rate-to-probability conversions
\item
  Enforce \(IR + BG_{mx} \leq 1\) constraint; flag and correct
  violations
\item
  Forecast secular trends in case fatality using ARIMA
\end{enumerate}

\begin{methodsnote}
Inline validation: \texttt{print(anyNA(data.out))} after every merge/join in \texttt{021\_get\_base\_rates.R}.
\end{methodsnote}
\end{frame}

\begin{frame}{CVD: Simulation Engine}
\phantomsection\label{cvd-simulation-engine}
\texttt{06\_run\_scenarios.R} (\textasciitilde84 KB) is the primary
simulation engine:

\begin{itemize}
\tightlist
\item
  Iterates over all age/sex/location/cause/year combinations
\item
  Applies intervention effects via coverage-adjusted relative risk
  reductions
\item
  Supports parallel execution across intervention combinations
\item
  Outputs: deaths averted, incident cases prevented, prevalent cases
\end{itemize}

\medskip

\textbf{Intervention stacking} applies effects sequentially to avoid
double-counting:

\begin{enumerate}
\tightlist
\item
  Sodium \(\rightarrow\) shifts BP distribution (population-level)
\item
  Statins \(\rightarrow\) reduces IR directly (individual-level)
\item
  TFA \(\rightarrow\) reduces IR via RR (population-level)
\item
  HTN control \(\rightarrow\) applies ETIHAD RR to treated fraction
\end{enumerate}

\end{frame}

\begin{frame}{CVD: DALY and Economic Outputs}
\phantomsection\label{cvd-daly-and-economic-outputs}
\textbf{Burden calculation} (\texttt{07\_output\_dalys.R}):

\begin{itemize}
\tightlist
\item
  YLL = deaths \(\times\) residual life expectancy (UNWPP 2024)
\item
  YLD = prevalent cases \(\times\) GBD disability weights
  (cause-specific)
\item
  DALYs = YLL + YLD, aggregated by age, sex, cause, year
\end{itemize}

\medskip

\textbf{Economic valuation}
(\texttt{08\_economic\_value\_calculation.R}, \textasciitilde42 KB):

\begin{itemize}
\tightlist
\item
  VSL calibrated to Indonesia GNI with income elasticity
  \(\varepsilon = 1.2\)
\item
  GDP growth from SSP2 projections (IIASA)
\item
  Outputs: monetized value of deaths averted, DALYs prevented, cost per
  DALY averted
\end{itemize}

\begin{frame}{Cancer: Three-Tier System}
\phantomsection\label{cancer-three-tier-system}
\textbf{16 cancer types} organized by analytic complexity:

\medskip

\textbf{Tier 1 --- Incidence/mortality only} (11 types): \small Uterine,
esophageal, thyroid, stomach, liver, pancreatic, ovarian, bladder,
lip/oral, nasopharynx, pharynx \normalsize

\medskip

\textbf{Tier 2 --- Staged} (3 types): \small Lung, prostate, breast
(local \(\rightarrow\) regional \(\rightarrow\) distant) \normalsize

\medskip

\textbf{Tier 3 --- Precancerous lesions modeled} (2 types): \small
Cervix, colorectal (well \(\rightarrow\) precancer \(\rightarrow\) local
\(\rightarrow\) regional \(\rightarrow\) distant) \normalsize

\begin{keyinsight}
Tier 3 cancers are the primary targets for screening interventions, where early detection shifts stage distribution and reduces late-stage mortality.
\end{keyinsight}
\end{frame}

\begin{frame}{Cancer: Core Markov Projection}
\phantomsection\label{cancer-core-markov-projection}
\texttt{project\_markov\_trace\_dt.R} implements the core engine:

\begin{itemize}
\tightlist
\item
  7\$\times\$7 transition probability matrix per (location, sex, cause,
  year, age)
\item
  State vector: \texttt{pwell, pprc, plcl, prgn, pdst, pcdx, pbgdx}
\item
  Forward projection via matrix multiplication, year-by-year
\item
  Two absorbing states: cancer death, background death
\item
  Handles newborn cohort initialization for population projection
\end{itemize}

\begin{methodsnote}
Transition probabilities are first computed at calibration years, then expanded to the full projection period via \texttt{expand\_tps\_years\_dt.R}. Matrix corrections enforced by \texttt{correct\_markov.R}.
\end{methodsnote}
\end{frame}

\begin{frame}{Cancer: Intervention Scenarios}
\phantomsection\label{cancer-intervention-scenarios}

\begin{itemize}

\begin{itemize}

\item
  Difference yields intervention-attributable outcomes
\end{itemize}

\begin{frame}{Comorbidity and Competing Risks}
\phantomsection\label{comorbidity-and-competing-risks}
\begin{cautionbox}[Not Modeled in v1.0]
CVD and cancer modules assume conditional independence. No comorbidity interaction or formal competing-risk adjustment is implemented.
\end{cautionbox}

\medskip

\textbf{Implication:}

\begin{itemize}
\tightlist
\item
  Deaths averted from one disease do not reduce the at-risk population
  for the other
\item
  Summing CVD + cancer results may overestimate total impact
\item
  Acceptable for single-disease policy evaluation; requires adjustment
  for combined NCD investment cases
\end{itemize}

\medskip

\textbf{Planned for v2.0:} Shared population with competing-risk
mortality across disease modules.
\end{frame}
